# Formato: (operando1 operando2 op_rel (expressao_true) (expressao_false) IF)

# Teste 1: Se 10 > 5, então RESULT1 = 100, senão RESULT1 = 0
# Deve imprimir 100
(10 5 > (100 RESULT1) (0 RESULT1) IF)

# Deve imprimir 1
