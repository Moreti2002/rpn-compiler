# Teste de Fibonacci - Versão Debug Ultra-Simples
# Calcula F(5) = 5 usando loop WHILE
# Demonstra funcionamento completo do algoritmo

# Inicializar variáveis
(0 A)
(1 B)
(5 N)

# Loop: enquanto N > 0, calcula próximo Fibonacci
(N 0 > (
    ((A B +) FIB)    # FIB = A + B
    (B A)            # A = B
    (FIB B)          # B = FIB
    ((N 1 -) N)      # N = N - 1
) WHILE)
