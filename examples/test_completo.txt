# ============================================================
# PROGRAMA DE TESTE COMPLETO - TODAS AS FUNCIONALIDADES
# ============================================================
# Este programa demonstra todas as funcionalidades implementadas
# nas Partes 1, 2 e 3 do compilador RPN
# ============================================================

# ============================================================
# SEÇÃO 1: Operações Aritméticas Básicas
# ============================================================
(2 3 +)
(10 5 -)
(4 7 *)
(20 4 /)
(17 5 %)
(2 8 ^)

# ============================================================
# SEÇÃO 2: Expressões Aninhadas
# ============================================================
((2 3 +) (4 5 *) +)
(((10 5 -) 2 *) (3 4 +) /)

# ============================================================
# SEÇÃO 3: Variáveis e Armazenamento
# ============================================================
(100 X)
(50 Y)
(X Y +)
(X Y *)

# ============================================================
# SEÇÃO 4: Memória (MEM)
# ============================================================
(42 MEMORIA)
(MEMORIA)
(MEMORIA 10 +)

# ============================================================
# SEÇÃO 5: Histórico (RES)
# ============================================================
(5 7 +)
(1 RES)
((1 RES) 10 *)

# ============================================================
# SEÇÃO 6: Operadores Relacionais
# ============================================================
(10 5 >)
(3 7 <)
(5 5 ==)
(8 3 !=)
(10 10 >=)
(4 9 <=)

# ============================================================
# SEÇÃO 7: Estruturas IF (Condicionais)
# ============================================================

# IF simples: se 10 > 5, então RESULTADO = 100, senão RESULTADO = 0
(10 5 > (100 RESULTADO) (0 RESULTADO) IF)

# IF com operações: se X >= Y, então Z = X+Y, senão Z = X-Y
(X Y >= ((X Y +) Z) ((X Y -) Z) IF)

# ============================================================
# SEÇÃO 8: Estruturas WHILE (Loops)
# ============================================================

# Inicializar contador
(0 CONTADOR)

# WHILE: enquanto CONTADOR < 5, incrementar CONTADOR
(CONTADOR 5 < ((CONTADOR 1 +) CONTADOR) WHILE)

# ============================================================
# SEÇÃO 9: Programa Completo - Cálculo de Fatorial
# ============================================================

# Inicializar variáveis
(5 NUM)
(1 FATORIAL)

# Loop simples: NUM > 1 → FATORIAL = FATORIAL * NUM
(NUM 1 > ((FATORIAL NUM *) FATORIAL) WHILE)

# ============================================================
# SEÇÃO 10: Expressões Complexas Combinadas
# ============================================================

# Combinação de todas as funcionalidades  
((X Y +) 2 *)
((X Y -) 3 /)
(100 50 >=)
(100 50 > (999 MAIOR) (111 MENOR) IF)
