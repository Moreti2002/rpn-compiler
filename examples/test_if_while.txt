# ============================================================
# TESTES COMPLETOS - ESTRUTURAS DE CONTROLE IF E WHILE
# ============================================================
# Este arquivo testa as estruturas de controle implementadas
# na Parte 3 do gerador TAC.
#
# Sintaxe RPN:
#   IF:    (operando1 operando2 op_rel (bloco_true) (bloco_false) IF)
#   WHILE: (operando1 operando2 op_rel (bloco) WHILE)
# ============================================================

# ============================================================
# TESTE 1: IF simples com números
# ============================================================
(5 3 > (10 X) (20 X) IF)
# Se 5 > 3 (verdadeiro), então X = 10, senão X = 20
# Esperado:
#   t0 = 5
#   t1 = 3
#   ifFalse t0 > t1 goto L1
#   t2 = 10
#   X = t2
#   goto L2
#   label L1
#   t3 = 20
#   X = t3
#   label L2

# ============================================================
# TESTE 2: IF com operações
# ============================================================
(10 5 >= ((2 3 +) Y) ((4 2 -) Y) IF)
# Se 10 >= 5, então Y = 2+3, senão Y = 4-2
# Esperado:
#   t0 = 10
#   t1 = 5
#   ifFalse t0 >= t1 goto L3
#   t2 = 2
#   t3 = 3
#   t4 = t2 + t3
#   Y = t4
#   goto L4
#   label L3
#   t5 = 4
#   t6 = 2
#   t7 = t5 - t6
#   Y = t7
#   label L4

# ============================================================
# TESTE 3: WHILE simples
# ============================================================
(10 0 > ((X 1 -) X) WHILE)
# Enquanto X > 0, faça X = X - 1
# Esperado:
#   label L5
#   t0 = 10
#   t1 = 0
#   ifFalse t0 > t1 goto L6
#   t2 = X
#   t3 = 1
#   t4 = t2 - t3
#   X = t4
#   goto L5
#   label L6

# ============================================================
# TESTE 4: WHILE com condição variável
# ============================================================
(N 10 < ((N 1 +) N) WHILE)
# Enquanto N < 10, faça N = N + 1
# Esperado:
#   label L7
#   t0 = N
#   t1 = 10
#   ifFalse t0 < t1 goto L8
#   t2 = N
#   t3 = 1
#   t4 = t2 + t3
#   N = t4
#   goto L7
#   label L8

# ============================================================
# TESTE 5: IF com comparação de igualdade
# ============================================================
(A B == (100 RESULT) (200 RESULT) IF)
# Se A == B, então RESULT = 100, senão RESULT = 200

# ============================================================
# TESTE 6: IF com comparação de desigualdade
# ============================================================
(X Y != (1 FLAG) (0 FLAG) IF)
# Se X != Y, então FLAG = 1, senão FLAG = 0

# ============================================================
# TESTE 7: Preparação para testes (atribuir variáveis)
# ============================================================
