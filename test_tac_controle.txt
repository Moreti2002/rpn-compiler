# ============================================================
# TESTE 1: Operadores relacionais básicos
# ============================================================
(5 3 >)
# Esperado:
# t0 = 5
# t1 = 3
# t2 = t0 > t1

# ============================================================
# TESTE 2: Operador menor que (<)
# ============================================================
(3 7 <)
# Esperado:
# t3 = 3
# t4 = 7
# t5 = t3 < t4

# ============================================================
# TESTE 3: Operador igual (==)
# ============================================================
(10 10 ==)
# Esperado:
# t6 = 10
# t7 = 10
# t8 = t6 == t7

# ============================================================
# TESTE 4: Operador diferente (!=)
# ============================================================
(5 8 !=)
# Esperado:
# t9 = 5
# t10 = 8
# t11 = t9 != t10

# ============================================================
# TESTE 5: Operador maior ou igual (>=)
# ============================================================
(10 10 >=)
# Esperado:
# t12 = 10
# t13 = 10
# t14 = t12 >= t13

# ============================================================
# TESTE 6: Operador menor ou igual (<=)
# ============================================================
(5 10 <=)
# Esperado:
# t15 = 5
# t16 = 10
# t17 = t15 <= t16

# ============================================================
# TESTE 7: Operadores relacionais com variáveis
# ============================================================
(10 X)
(5 Y)
(X Y >)
(X Y <)
(X Y ==)
# Esperado:
# t18 = 10
# X = t18
# t19 = 5
# Y = t19
# t20 = X
# t21 = Y
# t22 = t20 > t21
# t23 = X
# t24 = Y
# t25 = t23 < t24
# t26 = X
# t27 = Y
# t28 = t26 == t27

# ============================================================
# DEMONSTRAÇÃO: Como seria o TAC para IF (quando parser for atualizado)
# ============================================================
# Sintaxe RPN esperada: ((A B >) ((10)) ((20)) IF)
#
# TAC que seria gerado:
# t0 = A
# t1 = B
# t2 = t0 > t1      # condição
# ifFalse t2 goto L0
# t3 = 10           # bloco verdadeiro
# goto L1
# L0:
# t4 = 20           # bloco falso
# L1:
#
# ============================================================
# DEMONSTRAÇÃO: Como seria o TAC para WHILE (quando parser for atualizado)
# ============================================================
# Sintaxe RPN esperada: ((CONT 5 <) ((CONT 1 +) CONT) WHILE)
#
# TAC que seria gerado:
# L0:               # início do loop
# t0 = CONT
# t1 = 5
# t2 = t0 < t1      # condição
# ifFalse t2 goto L1
# t3 = CONT
# t4 = 1
# t5 = t3 + t4
# CONT = t5         # corpo do loop
# goto L0           # volta para início
# L1:               # fim do loop
